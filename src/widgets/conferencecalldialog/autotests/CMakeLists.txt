macro(add_ruqolaconferencecall_test _source)
    set(_test ${_source})
    get_filename_component(_name ${_source} NAME_WE )
    add_executable(${_name} ${_test} ${_name}.h)
    add_test(NAME ${_name} COMMAND ${_name} )
    ecm_mark_as_test(${_name})
    target_link_libraries(${_name} Qt::Test libruqolawidgets)
    set_target_properties(${_name} PROPERTIES DISABLE_PRECOMPILE_HEADERS ON)
endmacro()
add_ruqolaconferencecall_test(conferencecalldialogtest.cpp)
add_ruqolaconferencecall_test(conferencecallwidgettest.cpp)
add_ruqolaconferencecall_test(conferencedirectcalldialogtest.cpp)
add_ruqolaconferencecall_test(conferenceinfodialogtest.cpp)
add_ruqolaconferencecall_test(conferenceinfowidgettest.cpp)
